-- 신청 테이블
CREATE TABLE APPLY (
	APPLY_NO NUMBER NOT NULL,
	MOIM_NO VARCHAR2(10) NOT NULL,
	MEM_NO NUMBER NOT NULL,
	ACCEPT_YN CHAR(1) DEFAULT 'N',
	APPLY_DATE DATE	DEFAULT SYSDATE,
	ACCEPT_DATE DATE
);
-- 배너 테이블
CREATE TABLE BANNER (
	BANNER_NO NUMBER NOT NULL,
	MEM_NO NUMBER NOT NULL,
	CREATE_DATE DATE DEFAULT SYSDATE NOT NULL,
	ORIGIN_NAME VARCHAR2(500) NOT NULL,
	CHANGE_NAME VARCHAR2(500) NOT NULL,
	STATUS CHAR(1) DEFAULT 'Y'
);
-- 배틀 테이블
CREATE TABLE BATTLE (
	BATTLE_NO NUMBER NOT NULL,
	AREA VARCHAR2(10) DEFAULT '무관',
	GENDER CHAR(1) DEFAULT 'A',
	HOME_TEAM VARCHAR2(10),
	AWAY_TEAM VARCHAR2(10),
	"LEVEL"	CHAR(1)	DEFAULT 'E',
	BATTLE_DATE DATE,
	ORIGIN_NAME VARCHAR2(100),
	CHANGE_NAME VARCHAR2(100),
	STYLE VARCHAR2(10) NULL
);
-- 배틀 결과 테이블
CREATE TABLE BATTLE_RESULT (
	BATTLE_NO NUMBER NOT NULL,
	VICTORY VARCHAR2(10),
	DEFEAT VARCHAR2(10),
	VIC_RECORD VARCHAR2(200),
	DEF_RECORD VARCHAR2(200),
	OK CHAR(1) DEFAULT 'N'
);

CREATE TABLE BOARD (

	CREATE_DATE DATE DEFAULT SYSDATE,
	ORIGIN_NAME VARCHAR2(400),
	CHANGE_NAME VARCHAR2(400),
	COUNT NUMBER DEFAULT 0,
	STATUS CHAR(1) DEFAULT 'Y'
);

CREATE TABLE BOARD_REPLY (

);
