--  DDL for View VW_MEMBER
--------------------------------------------------------

  CREATE OR REPLACE FORCE VIEW "CONNECTIONPOOL"."VW_MEMBER" ("MEM_NO", "MEM_ID", "MEM_PWD", "GENDER", "INGIDO", "MEM_POINT", "EMAIL", "NICKNAME", "MEM_NAME", "MEM_AREA", "PHONE", "BIRTHDAY", "DETAIL_AREA", "MEM_STATUS", "ENROLL_DATE", "GRADE", "END_DATE", "TEAM_NO", "TEAM_GRADE", "FAILCNT") AS 
  SELECT 
		       MEM_NO,
		       MEM_ID,
		       MEM_PWD,
		       GENDER,
		       INGIDO,
		       MEM_POINT,
		       EMAIL,
		       NICKNAME,
		       MEM_NAME,
		       MEM_AREA,
		       PHONE,
		       TO_CHAR(BIRTHDAY, 'YYYY-MM-DD') as "BIRTHDAY",
		       DETAIL_AREA,
		       MEM_STATUS,
		       ENROLL_DATE,
		       GRADE,
		       END_DATE,
		       TEAM_NO,
		       TEAM_GRADE,
           FAILCNT
		  FROM 
		       MEMBER
     LEFT JOIN 
		       PENALTY USING (MEM_NO)
     LEFT JOIN
		       TEAM_MEMBER USING (MEM_NO)
         WHERE
                MEM_STATUS = 'Y'
;
